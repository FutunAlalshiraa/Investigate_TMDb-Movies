#calculate the correlation coefficient
def correlation_coe(x,y):
    C_x=(x-x.mean())/x.std(ddof=0)
    C_y=(y-y.mean())/y.std(ddof=0)
    return(C_x*C_y).mean()

correlation_coe(df['popularity'],df['profit'])
